    public static AccountType returnAcctType(String mytype){
        //CHECKING, SAVING, FIXED_DEPOSIT
        
        switch (mytype.toLowerCase()){
            case "c":
                return AccountType.CHECKING;
            case "s":
                return AccountType.SAVING;
            case "f":
                return AccountType.FIXED_DEPOSIT;
                
        }
        
        return AccountType.CHECKING;
    }
